// ConsoleHelper.cs - Beautiful console formatting
public static class ConsoleHelper
{
    public static void DisplayWorkflowProgress(BibWorkflowResult result)
    {
        Console.WriteLine("\nüìä BIB Workflow Execution:");
        Console.WriteLine($"   üÜî Workflow ID: {result.WorkflowId}");
        Console.WriteLine($"   üìç BIB.UUT.Port: {result.BibId}.{result.UutId}.{result.PortNumber}");
        Console.WriteLine($"   üì° Protocol: RS232");
        Console.WriteLine($"   üîå Physical Port: {result.PortName}");
        Console.WriteLine($"   ‚è±Ô∏è  Duration: {result.Duration.TotalSeconds:F2}s");
        Console.WriteLine();
        
        // 3-Phase execution details
        DisplayPhaseResult("üîã PowerOn Phase", result.StartResult);
        DisplayPhaseResult("üß™ Test Phase", result.TestResult);
        DisplayPhaseResult("üîå PowerOff Phase", result.StopResult);
        
        // Overall status
        Console.ForegroundColor = result.Success ? ConsoleColor.Green : ConsoleColor.Red;
        Console.WriteLine($"   üìã Overall Result: {(result.Success ? "‚úÖ SUCCESS" : "‚ùå FAILED")}");
        Console.ForegroundColor = ConsoleColor.White;
        
        if (!result.Success && !string.IsNullOrEmpty(result.ErrorMessage))
        {
            Console.ForegroundColor = ConsoleColor.Red;
            Console.WriteLine($"   ‚ùå Error: {result.ErrorMessage}");
            Console.ForegroundColor = ConsoleColor.White;
        }
    }
    
    private static void DisplayPhaseResult(string phaseName, PhaseResult? phase)
    {
        if (phase == null)
        {
            Console.WriteLine($"   {phaseName}: ‚è≠Ô∏è  Skipped");
            return;
        }
        
        var success = phase.SuccessfulCommands == phase.TotalCommands;
        var status = success ? "‚úÖ" : "‚ùå";
        
        Console.WriteLine($"   {phaseName}: {status} {phase.SuccessfulCommands}/{phase.TotalCommands} commands successful");
        
        // Show command details if any failed
        if (!success)
        {
            Console.ForegroundColor = ConsoleColor.Yellow;
            Console.WriteLine($"      ‚ö†Ô∏è  Failed commands: {phase.TotalCommands - phase.SuccessfulCommands}");
            Console.ForegroundColor = ConsoleColor.White;
        }
    }
}
